# Command Architecture + tauri-specta Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the three disconnected command execution paths in the editor (TypeScript registry, Rust registry, direct calls) with a single `EditorModule` + `CommandRegistry` system, unified `dispatchCommand` in TypeScript, and `tauri-specta`-generated bindings — eliminating all manually-maintained TypeScript type mirrors.

**Architecture:** A `CommandSpec`-based registry is owned by Rust, served to TypeScript via `list_commands`, and live-streamed to the future MCP server via a `watch::Receiver`. TypeScript module files register handlers for their commands; `dispatchCommand` routes locally if a handler exists or fires `editor-run-command` for Rust-only commands. All IPC types are generated from Rust structs at build time.

**Tech Stack:** Rust + Tauri 2, `specta` + `tauri-specta` (type generation), `tokio::sync::watch` (live catalog), `insta` (snapshot tests), SvelteKit + TypeScript (frontend), `xtask` (codegen + lint automation).

```rust
// engine/editor/src-tauri/bridge/registry.rs
use serde::Serialize;
use std::sync::Mutex;
use tokio::sync::watch;

/// Full descriptor for a single editor command.
#[derive(Debug, Clone, Serialize, specta::Type)]
pub struct CommandSpec {
    pub id: String,
    pub module_id: String,
    pub label: String,
    pub category: String,
    pub description: Option<String>,
    pub keybind: Option<String>,
    pub args_schema: Option<serde_json::Value>,
    pub returns_data: bool,
}

/// Implemented by every subsystem that exposes commands to the editor.
pub trait EditorModule: Send + Sync {
    fn id(&self) -> &str;
    fn commands(&self) -> Vec<CommandSpec>;
}

pub struct CommandRegistry {
    commands: Vec<CommandSpec>,
    module_index: std::collections::HashMap<String, Vec<usize>>,
    registry_tx: watch::Sender<Vec<CommandSpec>>,
}

impl CommandRegistry {
    /// Returns the registry and a watch receiver for live catalog updates.
    pub fn new() -> (Self, watch::Receiver<Vec<CommandSpec>>) {
        let (tx, rx) = watch::channel(Vec::new());
        let registry = Self {
            commands: Vec::new(),
            module_index: std::collections::HashMap::new(),
            registry_tx: tx,
        };
        (registry, rx)
    }

    /// Register all commands from a module. Sets `module_id` on each spec.
    /// Panics in debug builds if any command id doesn't start with `module.id() + "."`.
    pub fn register_module(&mut self, module: &dyn EditorModule) {
        let prefix = format!("{}.", module.id());
        let mut specs = module.commands();
        debug_assert!(
            !specs.is_empty(),
            "Module '{}' registered with no commands",
            module.id()
        );
        let base_idx = self.commands.len();
        let mut indices = Vec::with_capacity(specs.len());
        for (i, spec) in specs.iter_mut().enumerate() {
            debug_assert!(
                spec.id.starts_with(&prefix),
                "Command '{}' in module '{}' must start with '{}'",
                spec.id, module.id(), prefix
            );
            spec.module_id = module.id().to_string();
            indices.push(base_idx + i);
        }
        self.commands.extend(specs);
        self.module_index.insert(module.id().to_string(), indices);
        let _ = self.registry_tx.send(self.commands.clone());
    }

    pub fn list(&self) -> &[CommandSpec] {
        &self.commands
    }

    pub fn get(&self, id: &str) -> Option<&CommandSpec> {
        self.commands.iter().find(|c| c.id == id)
    }

    pub fn by_module(&self, module_id: &str) -> Vec<&CommandSpec> {
        match self.module_index.get(module_id) {
            Some(indices) => indices.iter().map(|&i| &self.commands[i]).collect(),
            None => Vec::new(),
        }
    }

    pub fn get_by_keybind(&self, keybind: &str) -> Option<&CommandSpec> {
        self.commands.iter().find(|c| {
            c.keybind.as_deref() == Some(keybind)
        })
    }
}
```

```rust
// engine/editor/src-tauri/bridge/modules/user_modules.rs
use std::collections::HashMap;
use std::sync::Mutex;
use crate::bridge::registry::CommandRegistry;

/// Trait for native handlers of user module commands.
pub trait ModuleCommandHandler: Send + Sync {
    fn run(&self, id: &str, args: Option<serde_json::Value>) -> Result<serde_json::Value, String>;
}

/// Tauri managed state: maps module_id to its native command handler.
pub struct UserModuleHandlerState(pub Mutex<HashMap<String, Box<dyn ModuleCommandHandler>>>);

impl UserModuleHandlerState {
    pub fn new() -> Self {
        UserModuleHandlerState(Mutex::new(HashMap::new()))
    }
}

/// Called from run_command_inner for commands with no built-in match.
/// For v1, returns Err("not implemented") for all user module commands.
pub fn run_module_command(
    _module_id: &str,
    _id: &str,
    _args: Option<serde_json::Value>,
    state: &UserModuleHandlerState,
) -> Result<serde_json::Value, String> {
    let handlers = state.0.lock().unwrap();
    match handlers.get(_module_id) {
        Some(handler) => handler.run(_id, _args),
        None => Err("not implemented".to_string()),
    }
}
```

```rust
// engine/editor/src-tauri/bridge/runner.rs
use serde::{Deserialize, Serialize};
use tauri::{AppHandle, Emitter};
use tracing::{error, warn};

use super::registry::{CommandRegistry, CommandSpec};

/// Known Rust-handled command ids (for v1: only screenshot).
const RUST_HANDLED: &[&str] = &["viewport.screenshot"];

/// Dispatches a Rust-handled command. Called by the editor-run-command listener
/// and (future) by the MCP AI bridge background task.
///
/// Returns `Ok(serde_json::Value::Null)` for fire-and-forget commands,
/// or `Ok(value)` for commands with `returns_data = true`.
pub fn run_command_inner(
    id: &str,
    args: Option<serde_json::Value>,
    registry: &CommandRegistry,
    app: &AppHandle,
) -> Result<serde_json::Value, String> {
    // Validate command exists in registry
    let spec = registry.get(id).ok_or_else(|| format!("Unknown command: {id}"))?;

    // NativeViewportState::capture_png_bytes() is responsible for returning Err on
    // unsupported platforms — platform guards live inside that method, not here.
    match id {
        "viewport.screenshot" => {
            use crate::viewport::NativeViewportState;
            let vp_state = app.state::<NativeViewportState>();
            let bytes = vp_state.capture_png_bytes()?;
            let b64 = base64_encode(&bytes);
            return Ok(serde_json::Value::String(b64));
        }
        other => {
            // Not a built-in Rust command — check if it's a user module command
            if !RUST_HANDLED.contains(&other) {
                // Attempt user module handler (returns Err("not implemented") for v1)
                use crate::bridge::modules::user_modules::run_module_command;
                use crate::bridge::modules::UserModuleHandlerState;
                let user_state = app.state::<UserModuleHandlerState>();
                let module_id = &spec.module_id;
                return run_module_command(module_id, other, args, &user_state);
            }
        }
    }

    Ok(serde_json::Value::Null)
}

#[derive(Deserialize)]
struct RunCommandPayload {
    id: String,
    args: Option<serde_json::Value>,
    #[serde(rename = "requestId")]
    request_id: Option<String>,
}

#[derive(Serialize, Clone)]
struct CommandResultPayload {
    id: String,
    #[serde(rename = "requestId")]
    request_id: String,
    result: Option<serde_json::Value>,
    error: Option<String>,
}

/// Register the editor-run-command Tauri event listener.
/// Call this once from lib.rs during app setup.
pub fn register_run_command_listener(
    app: &AppHandle,
    registry: std::sync::Arc<std::sync::Mutex<CommandRegistry>>,
) {
    let app_handle = app.clone();
    app.listen("editor-run-command", move |event| {
        let payload_str = event.payload();
        let payload: RunCommandPayload = match serde_json::from_str(payload_str) {
            Ok(p) => p,
            Err(e) => {
                warn!(error = %e, "editor-run-command: failed to parse payload");
                return;
            }
        };

        let id = &payload.id;

        // Only handle known Rust-side commands — log and drop everything else
        if !RUST_HANDLED.contains(&id.as_str()) {
            warn!(
                command_id = %id,
                "editor-run-command: not a Rust-handled command (programming error: TS handler missing)"
            );
            return;
        }

        let result = {
            let registry_guard = registry.lock().unwrap();
            run_command_inner(id, payload.args.clone(), &registry_guard, &app_handle)
        };

        // Only emit result event if the caller provided a requestId (returns_data = true path)
        if let Some(request_id) = payload.request_id {
            let result_payload = match result {
                Ok(value) => CommandResultPayload {
                    id: id.clone(),
                    request_id,
                    result: Some(value),
                    error: None,
                },
                Err(e) => {
                    error!(command_id = %id, error = %e, "editor-run-command: execution failed");
                    CommandResultPayload {
                        id: id.clone(),
                        request_id,
                        result: None,
                        error: Some(e),
                    }
                }
            };
            let _ = app_handle.emit("editor-command-result", result_payload);
        } else if let Err(e) = result {
            error!(command_id = %id, error = %e, "editor-run-command: fire-and-forget failed");
        }
    });
}
```

```typescript
import type { CommandSpec } from '$lib/bindings';
import { listen } from '@tauri-apps/api/event';
import { commands } from '$lib/bindings';

// --- Registry snapshot ---
const _registry = new Map<string, CommandSpec>();

export function populateRegistry(specs: CommandSpec[]): void {
  _registry.clear();
  for (const spec of specs) {
    _registry.set(spec.id, spec);
  }
}

// --- Handler table ---
const _handlers = new Map<string, (args?: Record<string, unknown>) => unknown | Promise<unknown>>();

export function registerCommandHandler(
  id: string,
  fn: (args?: Record<string, unknown>) => unknown | Promise<unknown>
): void {
  _handlers.set(id, fn);
}

// --- Dispatch ---
export async function dispatchCommand(
  id: string,
  args?: Record<string, unknown>
): Promise<unknown> {
  const spec = _registry.get(id);
  if (!spec) throw new Error(`Unknown command: ${id}`);

  const handler = _handlers.get(id);
  if (handler) {
    const result = await handler(args);
    return spec.returns_data ? result : undefined;
  }

  // No TS handler — fire event to Rust
  if (!spec.returns_data) {
    await import('@tauri-apps/api/event').then(({ emit }) =>
      emit('editor-run-command', { id, args })
    );
    return undefined;
  }

  // returns_data=true Rust command — await result via requestId
  const requestId = crypto.randomUUID();
  return new Promise((resolve, reject) => {
    const TIMEOUT_MS = 5000;
    let unlisten: (() => void) | undefined;
    const timer = setTimeout(() => {
      unlisten?.();
      reject(new Error(`Command timed out: ${id}`));
    }, TIMEOUT_MS);

    listen<{ id: string; requestId: string; result?: unknown; error?: string }>(
      'editor-command-result',
      (event) => {
        if (event.payload.requestId !== requestId) return;
        unlisten?.();
        clearTimeout(timer);
        if (event.payload.error) {
          reject(new Error(event.payload.error));
        } else {
          resolve(event.payload.result);
        }
      }
    ).then((fn) => {
      unlisten = fn;
      import('@tauri-apps/api/event').then(({ emit }) =>
        emit('editor-run-command', { id, args, requestId })
      );
    });
  });
}

/** Test-only: reset internal state. */
export function _resetForTesting(): void {
  _registry.clear();
  _handlers.clear();
}
```

```rust
    // Extract the function body by counting braces
    let after_fn = &source[start..];
    let body_start = after_fn
        .find('{')
        .ok_or_else(|| anyhow::anyhow!("handleKeyDown has no opening brace"))?;
    let body = extract_brace_block(&after_fn[body_start..]);

    // Find all identifier( call patterns
    let re = regex::Regex::new(r"\b([a-zA-Z_][a-zA-Z0-9_]*(?:\.[a-zA-Z_][a-zA-Z0-9_]*)*)\s*\(")?;
    for cap in re.captures_iter(body) {
        let callee = &cap[1];
        // Skip allowed calls
        if ALLOWED_CALLS.iter().any(|&a| callee.contains(a)) {
            continue;
        }
        anyhow::bail!(
            "xtask lint: handleKeyDown calls '{}()' — only dispatchCommand and getByKeybind are allowed",
            callee
        );
    }

    println!("xtask lint: handleKeyDown OK");
    Ok(())
}

fn extract_brace_block(s: &str) -> &str {
    let mut depth = 0;
    let mut end = 0;
    for (i, c) in s.char_indices() {
        match c {
            '{' => depth += 1,
            '}' => {
                depth -= 1;
                if depth == 0 {
                    end = i + 1;
                    break;
                }
            }
            _ => {}
        }
    }
    &s[..end]
}
```
